def powersTen(quantity:int)->str:
    '''
    return the powers of ten of the input
    :param quantity:
    :return:
    '''
    out = ""
    sufix = ["unit", "kilo", "mega", "giga", "tera"]
    for power in range(5):
        value = f"{quantity} {'0'*3*power}".ljust(25) + sufix[power]
        out += value + "\n"

    return out
